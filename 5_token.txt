function encoding = customWordEncoding(documents, varargin)
    % CUSTOMWORDENCODING Creates a text encoding object from document data.
    %
    % Syntax:
    %   encoding = customWordEncoding(documents)
    %   encoding = customWordEncoding(documents, 'UnknownToken', "<unk>", 'IgnoreCase', true)
    
    % 1. Parse optional inputs
    p = inputParser;
    addRequired(p, 'documents');
    addParameter(p, 'UnknownToken', "<unknown>", @(x) isstring(x) || ischar(x));
    addParameter(p, 'IgnoreCase', true, @islogical);
    parse(p, documents, varargin{:});
    
    unkToken = string(p.Results.UnknownToken);
    ignoreCase = p.Results.IgnoreCase;

    % 2. Tokenize text and extract unique vocabulary
    allTokens = tokenizeText(documents, ignoreCase);
    uniqueTokens = unique(allTokens, 'stable');
    
    % Ensure the unknown token sits cleanly at index 1
    uniqueTokens(uniqueTokens == unkToken) = [];
    vocabulary = [unkToken, uniqueTokens];
    
    % 3. Create key-value lookup containers
    indices = 1:numel(vocabulary);
    wordToIdxMap = containers.Map(vocabulary, indices);
    idxToWordMap = containers.Map(indices, vocabulary);
    
    % 4. Build the structural encoding package
    encoding = struct();
    encoding.Vocabulary = vocabulary;
    encoding.UnknownToken = unkToken;
    encoding.NumWords = numel(vocabulary);
    
    % 5. Attach functional encoding handles
    encoding.doc2sequence = @(docs) encodeDocs(docs, wordToIdxMap, unkToken, ignoreCase);
    encoding.sequence2doc = @(seq) decodeSequence(seq, idxToWordMap, unkToken);
end

%% Helper: Document Tokenizer
function tokens = tokenizeText(textData, ignoreCase)
    combined = join(string(textData), " ");
    if ignoreCase
        combined = lower(combined);
    end
    % Strip common punctuation marks
    cleaned = regexprep(combined, '[.,!?;:()\[\]"''\-\\]', ' ');
    % Split words by whitespace boundaries
    tokens = split(normalizeSpace(cleaned));
    tokens = tokens(tokens ~= "")';
end

%% Helper: Word to Numeric Index Encoder
function sequence = encodeDocs(textData, mapObj, unkToken, ignoreCase)
    tokens = tokenizeText(textData, ignoreCase);
    sequence = zeros(1, numel(tokens));
    unkIdx = mapObj(unkToken);
    
    for i = 1:numel(tokens)
        token = tokens(i);
        if isKey(mapObj, token)
            sequence(i) = mapObj(token);
        else
            sequence(i) = unkIdx;
        end
    end
end

%% Helper: Numeric Index to Word Decoder
function tokens = decodeSequence(sequence, mapObj, unkToken)
    tokens = repmat("", 1, numel(sequence));
    for i = 1:numel(sequence)
        idx = sequence(i);
        if isKey(mapObj, idx)
            tokens(i) = mapObj(idx);
        else
            tokens(i) = unkToken;
        end
    end
end
